#include <iostream>
#include <vector>  // store arrays
#include <functional> // lambda function
#include <queue> // create priority queue
#include <utility> // store pair in priority queue
using namespace std;

void solve() {
    int n, k; // n vertices, k colors
    cin >> n >> k;
    int u, v; // u (start vertex) - edge - v (end vertex)
    
    cout << "DEBUG" << 1 << endl;

    //Build graph (adjacency list)
    vector<vector <int>> adj(n);
    for (int i = 0; i < n; i++) {
        cin >> u >> v;
        u--;
        v--;
        adj[u].push_back (v);
        adj[v].push_back (u);
    }

    cout << "DEBUG" << 2 << endl;

    //Build color array 
    vector <int> C(k);
    for (int i = 0; i < k; i ++) {
        cin >> C[i];
    }

    cout << "DEBUG" << 3 << endl;

    // Special case N = 2
    if (n == 2) {
        int col = -1;
        for (int i = 0; i < k; i++) {
            if (C[i] >= 2) { col = i + 1; break; }
        }
        if (col == -1) cout << "-1\n";
        else cout << col << " " << col << "\n";
        return;
    }    

    // Count number of leaves 
    int L = 0;
    for (int i = 0; i < n; i++) {
        if (adj[i].size() == 1) {
            L++;
        }
    }

    cout << "DEBUG" << 4 << endl;

    //Build tree structure using DFS
    vector <int> sub_leaves(n, 0);
    vector <int> order;
    vector <int> parent (n, -1);

    int root = -1;
    for (int i = 0; i < n; i++) {
        if (adj[i].size() > 1) {
            root = i; 
            break;
        }
    }

    cout << "DEBUG" << 5;
    cout << endl;
    
    function<void(int, int)> dfs = [&](int v, int p) {
        parent[v] = p;
        bool isleaf = true;
        for (int u : adj[v]) {
            if (u == p) {continue;}
            isleaf = false;
            parent[u] = v;
            dfs(u, v);
            sub_leaves[v] += sub_leaves[u];
        }
        if (isleaf) sub_leaves[v] = 1;
        order.push_back(v);
    };
    dfs (root, -1);

    cout << "DEBUG" << 6;
    cout << endl;    

    // Find leaf centroid
    /*
    Leaf centroid has to satisfy: 
        - Divide upper part with <= L / 2 leaves
        - Divide other adj part with <= L / 2 leaves
    */
    int x = -1; // leaf centroid
    for (int v : order) {
        if (adj[v].size() == 1){continue;}
        //Check # of leaves of upper part
        bool satisfy = ((L - sub_leaves[v]) <= L / 2);
        if (!satisfy) {continue;} 
        for (int u : adj[v]) {
            if (u == parent[v]) {continue;}
            if (sub_leaves[u] > L / 2) {
                satisfy = false; 
                break; 
            }
        }
        if (satisfy){
            x = v; 
            break;
        }
    }

    cout << "DEBUG" << 7;
    cout << endl;    

    // Collect leaves into groups (DFS to group)
    vector<vector <int>> groups;
    for (int start : adj[x]) {
        vector <int> leaves;
        function<void(int, int)> collect = [&](int v, int p){
            if (adj[v].size() == 1) {
                leaves.push_back(v);
            }
            for (int u : adj[v]){
                if (u == p) {continue;}
                collect (u, v);
            }
        };
        collect (start, x);
        if (!leaves.empty()){
            groups.push_back(leaves);            
        }
    }
    int G = groups.size();

    cout << "DEBUG" << 8;
    cout << endl;        

    //Create priority queue for each group
    vector <int> ptr(G, 0); //store the index of vertex in each group
    priority_queue<pair <int, int>> pq; // store number of vertices in each group and group id
    for (int g = 0; g < G; g++) {
        pq.push ({(int)groups[g].size(), g});
    }

    cout << "DEBUG" << 9;
    cout << endl;        

    // Color leaves
    /*
    Helper function: 
        - paint_leave:
            - paint each vertex in each leave groups with a color 
            - udpate the unpainted leaves in each group (priority queue)

        - get_top: 
            - get the top position of the priority queue
            - update the number of unpainted leaves

    While there is remaining color and unpainted leaves: 
    rem = remaining slot of the current color

    Case 1: one remaining leave
        - Color the last leave with the current color
        - Color centroid with the same color 

    Case 2: remaining color is enough for remaining leaves
        - Spam paint_leaf and get_top to color all remaining leaves

    Case 3: Not the two above  cases 
        - Get the two groups with most leaves unpainted
        - Paint the two groups with the current color
        - Repeat the two step above until it either meets case 1 or 2, or run out of color
    */
    int unpainted = L; //number of unpainted leaves 
    vector <int> color (n, -1);  //the returning color array 

    cout << "DEBUG" << 10;
    cout << endl;        

    auto paint_leaf = [&](int g, int c){
        int v = groups[g][ptr[g]++];
        color[v] = c;
        unpainted--;
    };

    auto get_top = [&] () -> pair <int, int>{
        while (!pq.empty()) {
            auto [cnt, g] = pq.top();
            int actual_count = (int)groups[g].size() - ptr[g];
            if (actual_count == 0) {
                pq.pop();
            }
            else if (actual_count != cnt) {
                pq.pop();
                pq.push ({actual_count, g});
            }
            else {
                return pq.top();
            }
        }
        return {-1, -1};
    };

    for (int i = 0; i < k; i++){
        int rem = C[i]; 
        while (rem > 0 && unpainted > 0) {
            //Case 1: 1 last unpainted leaf 
            /*
            - Get the last unpainted leaf
            - Paint it with the current color 
            - Paint centroid with the same color
            */
           if (unpainted == 1) {
                cout << "DEBUG" << 11;
                cout << endl;                        
                for (int g = 0; g < G; g++){
                    if (ptr[g] < (int)groups[g].size()) {
                        paint_leaf (g, i + 1);
                        break;
                    }
                }
                if (color[x] == -1) {
                    color[x] = i + 1;
                }
                rem = 0;
                break;
            }
            // Case 2: remaining color is enough for remaining leaves
            /*
            - Spamming paint_leaf and get_top to paint the leaves
            - Pop and push the pq along the way
            */
            if (rem >= unpainted) {
                while (unpainted > 0) {
                    auto [cnt, g] = get_top();
                    paint_leaf (g, i + 1);
                    rem --;
                    pq.pop();
                    if (ptr[g] < (int)groups[g].size()) {
                        pq.push({groups[g].size() - ptr[g], g});
                    }
                }
                break;
            }

            //Case 3: not the above two cases 
            /*
            - Pop the two groups with most number of unpainted leaves 
            - Paint them with the current color
            - Spamming that color to shrink the size of the largest unpainted group
            */
            else {

                //Paint the two groups with most unpainted leaves
                auto[cnt1, g1] = get_top();
                paint_leaf(g1, i + 1);
                pq.pop();
                auto[cnt2, g2] = get_top();
                paint_leaf(g2, i + 1);
                pq.pop();

                rem -= 2;     
                
                if (ptr[g1] < (int)groups[g1].size()) {
                    pq.push ({groups[g1].size() - ptr[g1], g1});
                }

                if (ptr[g2] < (int)groups[g2].size()) {
                    pq.push ({groups[g2].size() - ptr[g2], g2});
                }           

                //Greedy: Shrink the size of the groups with current most # of unpainted leaves using the current color
                while (rem > 0 && unpainted > 0) {
                    auto[cnt, g] = get_top();
                    if (g == -1) break;
                    paint_leaf(g, i + 1);
                    pq.pop();
                    rem --;
                    if (ptr[g] < (int)groups[g].size()){
                        pq.push ({groups[g].size() - ptr[g], g});                        
                    }
                }
            }
        }
        if (unpainted > 0) {
            cout << -1 << endl;
            return;
        }
    }

    // Color the remaining vertices
    /*
    2 pointers
    - 1st pointer will track each vertex
    - 2nd pointer will track the color 
    */
    
    //set up used colors array
    vector <int> used (k, 0);
    for (int v = 0; v < n; v++) {
        if (color[v] == -1) {continue;}
        used[color[v] - 1]++;
    }
    
    int ci = 0;//2nd pointer
    //1st pointer
    for (int v = 0; v < n; v++) {
        //Skip colored vertices
        if (color[v] != -1) {continue;}
        //Skip unavailable color
        while (used[ci] >= C[ci] && ci < k) {ci++;}
        color[v] = ci + 1;
        used[ci]++;
    }

    for (int i = 0; i < n; i++) {
        cout << color[i];
        if (i != n - 1) {cout << " ";}
    }
    cout << endl;
}

int main() {
    int t; cin >> t;
    for (int i = 0; i < t; i++){
        solve();
    }
}
